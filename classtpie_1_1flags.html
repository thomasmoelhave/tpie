Detailed Description

template<typename T>
class tpie::flags< T >

Type safe bitflags over an enumeration T.

Example:

enum OPTION {a=1, b=2, c=4};
TPIE_DECLARE_OPERATORS_FOR_FLAGS(OPTION)
typedef tpie::flags<OPTION> OPTIONS;
void foo(OPTIONS o) {
if (o & a) do_something();
}
void bar() {
foo(a | c);
}
